⚙️ How Django Forms Work
# forms.py
class ContactForm(forms.Form):
    name = forms.CharField(max_length=100)
    email = forms.EmailField()
    message = forms.CharField(
        widget=forms.Textarea,
        min_length=10
    )
    
    # Custom validation
    def clean_email(self):
        email = self.cleaned_data['email']
        if '@spam.com' in email:
            raise forms.ValidationError(
                "Spam not allowed!"
            )
        return email  # Always return!
🔄 PRG Pattern (Post-Redirect-Get)
# views.py
def form_demo(request):
    if request.method == 'POST':
        form = ContactForm(request.POST)
        if form.is_valid():
            # Process the data...
            name = form.cleaned_data['name']
            
            # PRG: redirect after POST
            # Prevents duplicate submission on F5
            messages.success(request, "Sent!")
            return redirect('form_demo')
        # If invalid: fall through to render
        # The form contains error messages
    else:
        form = ContactForm()  # Empty form
    
    return render(request, 'form.html',
                  {'form': form})
🛡️ CSRF Protection

Every POST form needs {% verbatim %}{% csrf_token %}{% endverbatim %}.

Django generates a random token, embeds it in the form, and verifies it matches when the form is submitted. This prevents Cross-Site Request Forgery attacks.
