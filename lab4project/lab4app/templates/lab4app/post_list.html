# Filter: WHERE status = 'published'
posts = Post.objects.filter(status='published')

# Search with Q objects (OR conditions)
posts = posts.filter(
    Q(title__icontains=query) |
    Q(content__icontains=query)
)

# Annotate: add COUNT as computed column
posts = posts.annotate(
    comment_count=Count('comments')
)

# select_related: JOIN to avoid N+1 queries
posts = posts.select_related('author')

# prefetch_related: for ManyToMany
posts = posts.prefetch_related('categories')
